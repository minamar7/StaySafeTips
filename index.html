async function startQuiz(mode = 'free') { // Το ./ διασφαλίζει ότι ψάχνουμε στον φάκελο /StaySafeTips/ const fileName = `./questions_${mode}_${currentLang}.json`; console.log("Φόρτωση αρχείου:", fileName); try { const response = await fetch(fileName); if (!response.ok) { throw new Error(`Δεν βρέθηκε το αρχείο: ${fileName}`); } const data = await response.json(); // Επιλογή 5 τυχαίων ερωτήσεων quiz_data = data.sort(() => Math.random() - 0.5).slice(0, 5); current_q = 0; nav('quiz'); showQuestion(); } catch (err) { console.error("Σφάλμα:", err); alert("Πρόβλημα σύνδεσης: " + err.message); } }
